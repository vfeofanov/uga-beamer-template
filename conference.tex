\begin{frame}{Mean value theorem}
\begin{thm}
Let  $f\,:\,[a,b]\to \mathbb{R}$ be a continuous function on $[a,b]$, and differentiable on $(a,b)$ with $a<b$. Then there exists $c\in (a,b)$ such that
\[
f'(c) = \frac{f(b)-f(a)}{b-a}.
\]
\end{thm}
\end{frame}
